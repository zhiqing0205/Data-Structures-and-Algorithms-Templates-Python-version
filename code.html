    class StringHash:
        # 字符串哈希，用O(n)预处理，O(1)获取一段哈希值
        def __init__(self, s):
            n = len(s)
            self.BASE = BASE = 131  # 进制 131
            self.MOD = MOD = 2 ** 64
            self.h = h = [0] * (n + 1)
            self.p = p = [1] * (n + 1)
            for i in range(1, n + 1):
                p[i] = (p[i - 1] * BASE) % MOD
                h[i] = (h[i - 1] * BASE % MOD + ord(s[i - 1])) \\
                                                            % MOD
        
        # 用O(1)时间获取开区间[l,r)（即s[l:r]）的哈希值
        def get_hash(self, l, r):
            return (self.h[r] - self.h[l] * self.p[r - l] % \\
                                    self.MOD) % self.MOD

    def Floyd(graph):
        n = len(graph)
        # 初始化距离矩阵
        dist = [[float('inf') for _ in range(n)] \\
                                    for _ in range(n)]
        # 将图中的边的权值赋值给距离矩阵
        for i in range(n):
            for j in range(n):
                dist[i][j] = graph[i][j]
        # Floyd算法
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    dist[i][j] = min(dist[i][j], dist[i][k] \\
                                                + dist[k][j])
        return dist

    def merge_sort(arr,left,right):       
        if left == right:        
            return
        # 将较大的区间[left,right]一分为二
        mid=(left + right) >> 1          
        merge_sort(arr, left, mid)
        merge_sort(arr, mid+1, right)
        i, j=left, mid+1   # i,j分别记录两个区间的起点
        tmp = []           # tmp记录[left,right]的排序结果
        while i <= mid or j <= right:
            if i > mid or (j <= right and arr[i] > arr[j]):
                tmp.append(arr[j])
                j += 1    
            else:
                tmp.append(arr[i])
                i += 1
        arr[left:right + 1]=tmp[:]  # tmp的记录结果传回arr 
    merge_sort(arr, 0, len(arr) - 1)

    class Trie:
        def __init__(self):
            self.children = [None] * 26
            self.isEnd = False
        
        # 判断是否存在以'prefix'为前缀的单词
        def searchPrefix(self, prefix: str) -> "Trie":
            node = self
            for ch in prefix:
                ch = ord(ch) - ord("a")
                if not node.children[ch]:
                    return None
                node = node.children[ch]
            return node
        
        # 插入单词
        def insert(self, word: str) -> None:
            node = self
            for ch in word:
                ch = ord(ch) - ord("a")
                if not node.children[ch]:
                    node.children[ch] = Trie()
                node = node.children[ch]
            node.isEnd = True

    def bfs(v):
        v.visited = True
        queue = []  # 初始化一个空队列
        queue.append(v)  # v 入队
        while queue:
            for w in v.neighbours:
                if not w.visited:
                    w.visited = True
                    queue.append(w)
            queue.pop(0)  # v 出队

    def isPrime(x):
        if x < 2:
            return False
        # 从2到sqrt(x)判断是否有因子
        n = int(sqrt(x))
        for i in range(2, n + 1):
            if x % i == 0:
                return False
        return True

    # Dijkstra + 堆优化
    def Dijkstra_heap(graph, s):
        n = len(graph)
        # 初始化距离矩阵
        dist = [float('inf') for _ in range(n)]
        # 初始化标记数组
        vis = [False for _ in range(n)]
        # 将源点的距离设为0
        dist[s] = 0
        # 初始化堆
        heap = []
        heappush(heap, (0, s))
        # Dijkstra算法
        while heap:
            # 找到距离最小的点
            d, u = heappop(heap)
            # 如果该点已经标记过，跳过
            if vis[u]:
                continue
            # 标记该点
            vis[u] = True
            # 更新距离
            for v in range(n):
                if dist[v] > dist[u] + graph[u][v]:
                    dist[v] = dist[u] + graph[u][v]
                    heappush(heap, (dist[v], v))
        return dist

    def topologicalSort(self,items,indegree,neighbors):
        queue = deque()
        res = []
        
        # 将入度为0的节点加入队列
        for item in items:
            if indegree[item]==0:
                queue.append(item)
        
        # 每次从队列中取出一个节点，将其邻接节点的入度减1
        while queue:
            cur = queue.popleft()
            res.append(cur)

            for neighbor in neighbors[cur]:
                indegree[neighbor] -= 1
                if indegree[neighbor] == 0:
                    queue.append(neighbor)
                    
        return res
